141.lINKED LIFE CYCLE


public class Solution {
    public boolean hasCycle(ListNode head) {

        if(head == null||head.next == null) return false;
        ListNode slow = head;
        ListNode fast = head;
        // boolean flag = false;
        while(
            fast!=null && fast.next != null 
        ){
            slow = slow.next;
            fast = fast.next.next;
            if( slow == fast) {
                return true;
            } 
        }
        return false;

    }
}
